    function getReadiness(chName) {

        const data = db.chapters[chName] || {

            lecture: 0, dppDone: 0, dppTotal: 0, dppAcc: 0, level: 0, 

            revisedDate: Date.now(), revCount: 0, traps: 0, 

            aiClarity: 0, aiOral: 0, mockScore: 0

        };


        // 1. Decay Constant (Half-life of knowledge)

        const daysSinceRev = (Date.now() - data.revisedDate) / (1000 * 60 * 60 * 24);

        const decay = Math.exp(-0.015 * daysSinceRev); // 1.5% daily decay


        // 2. Mistake Impact

        const chMistakes = db.mistakes.filter(m => m.chapter === chName).length;

        const mistakePenalty = Math.min(30, chMistakes * 5);


        // 3. Main Readiness Formula

        let main = (

            (data.lecture * 10) +

            (data.dppAcc * 0.20) +

            (data.mockScore * 0.30) +

            (data.level >= 2 ? 15 : 5) +

            (data.revCount * 5) +

            (data.traps * 5)

        ) * decay;


        // 4. Advanced Readiness Formula

        let adv = (

            (data.aiOral * 0.25) +

            (data.aiClarity * 0.20) +

            (data.level >= 3 ? 20 : 0) +

            (data.dppAcc * 0.15) - 

            mistakePenalty

        ) * decay;


        // Caps

        if (data.level < 2) main = Math.min(main, 45); 

        if (data.level < 3) adv = Math.min(adv, 30);


        return { main: Math.max(0, Math.round(main)), adv: Math.max(0, Math.round(adv)) };

    }

    function updateDashboard() {

        let mainSum = 0, advSum = 0, count = 0;

        for (let ch in db.chapters) {

            const r = getReadiness(ch);

            mainSum += r.main;

            advSum += r.adv;

            count++;

        }

        

        const mainAvg = count > 0 ? Math.round(mainSum / count) : 0;

        const advAvg = count > 0 ? Math.round(advSum / count) : 0;
